    2. **Create `fmai-nextjs/src/app/[locale]/opengraph-image.tsx`** — Root-level locale OG image (serves as homepage image + fallback for any page without its own). Must export `alt`, `size`, `contentType`, and a default async function `Image`.

    Use `ImageResponse` from `next/og` (NOT `next/server`). Load font via `readFile` + `join(process.cwd(), 'public/fonts/ACTUAL_FILENAME.ttf')`. Use `getTranslations` from `next-intl/server` to get the homepage title/description from the `home` translation namespace. Pass to `OgImageTemplate`.

    ```typescript
    import { ImageResponse } from 'next/og'
    import { getTranslations } from 'next-intl/server'
    import { readFile } from 'node:fs/promises'
    import { join } from 'node:path'
    import { OgImageTemplate } from '@/lib/og-image'

    export const alt = 'Future Marketing AI — AI-Powered Marketing Automation'
    export const size = { width: 1200, height: 630 }
    export const contentType = 'image/png'

    export default async function Image({ params }: { params: Promise<{ locale: string }> }) {
      const { locale } = await params
      const t = await getTranslations({ locale, namespace: 'home' })
      const fontData = await readFile(join(process.cwd(), 'public/fonts/ACTUAL_FILENAME.ttf'))

      return new ImageResponse(
        <OgImageTemplate
          title={t('meta.title')}
          description={t('meta.description')}
          label="Future Marketing AI"
        />,
        {
          ...size,
          fonts: [{ name: 'DM Sans', data: fontData, weight: 600, style: 'normal' }],
        }
      )
    }
    ```
