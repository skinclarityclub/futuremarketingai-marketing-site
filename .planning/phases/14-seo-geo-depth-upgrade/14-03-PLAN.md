### Updated component

```typescript
import type { WithContext, Article } from 'schema-dts'
import { JsonLd } from './JsonLd'
import {
  SITE_URL,
  SITE_NAME,
  ORG_ID,
  DALEY_PERSON_ID,
} from '@/lib/seo-config'

interface ArticleJsonLdProps {
  title: string
  description: string
  author: string
  datePublished: string
  dateModified: string
  slug: string
  locale: string
  /** Optional absolute image URL. Falls back to og-image.png. 1200x630 recommended. */
  image?: string
}

export function ArticleJsonLd({
  title,
  description,
  author,
  datePublished,
  dateModified,
  slug,
  locale,
  image,
}: ArticleJsonLdProps) {
  const url = `${SITE_URL}/${locale}/blog/${slug}`
  const articleId = `${url}#article`
  const imageUrl = image ?? `${SITE_URL}/og-image.png`

  const authorRef =
    author === 'Daley Maat' || author === 'Daley'
      ? { '@id': DALEY_PERSON_ID }
      : {
          '@type': 'Person' as const,
          name: author,
          url: SITE_URL,
        }

  const data: WithContext<Article> = {
    '@context': 'https://schema.org',
    '@type': 'Article',
    '@id': articleId,
    headline: title,
    description,
    datePublished,
    dateModified,
    author: authorRef,
    publisher: {
      '@type': 'Organization',
      '@id': ORG_ID,
      name: SITE_NAME,
      url: SITE_URL,
      logo: {
        '@type': 'ImageObject',
        url: `${SITE_URL}/og-image.png`,
        width: 1200,
        height: 630,
      },
    },
    image: {
      '@type': 'ImageObject',
      url: imageUrl,
      width: 1200,
      height: 630,
    },
    mainEntityOfPage: {
      '@type': 'WebPage',
      '@id': `${url}#webpage`,
    },
    url,
    inLanguage: locale,
    isPartOf: {
      '@type': 'WebSite',
      '@id': `${SITE_URL}/#website`,
    },
  }

  return <JsonLd data={data} />
}
```

### Notes

- `image` defaults to `/og-image.png` (existing since Phase 10). Blog posts can override with a post-specific cover image via frontmatter (future enhancement).
- `publisher.logo` is now correctly an `ImageObject` with `width`/`height` — Google flags missing logo dimensions as a warning.
- `mainEntityOfPage` references the WebPage @id emitted by WebPageJsonLd (14-02 added WebPage @id). Cross-reference coherence.
- `@id` on Article itself uses the article URL + `#article` fragment for graph linking.
