## Design Tokens for Cards

All cards share a base pattern:

- Container: `bg-bg-elevated/80 backdrop-blur-md border border-border-primary rounded-xl p-4`
- Hover: `hover:border-accent-system/30 transition-colors duration-200`
- Labels: `text-xs text-text-secondary font-mono uppercase tracking-wider`
- Values: `text-sm text-text-primary font-sans`
- Data values: `font-mono text-accent-system` (teal for key metrics)
- CTA links: `text-accent-system hover:underline text-xs`
- Width: `w-full` (fills message bubble width)

**ToolResultRenderer** — main router component:

```tsx
interface ToolResultRendererProps {
  part: any // ToolInvocationUIPart from AI SDK — use any to avoid tight coupling to internal types
}

export function ToolResultRenderer({ part }: ToolResultRendererProps) {
  // Loading states
  if (part.state === 'input-streaming' || part.state === 'input-available') {
    return <ToolLoadingCard toolName={part.toolName} />
  }
  // Error state
  if (part.state === 'output-error') {
    return <ToolErrorCard />
  }
  // Output available — route to correct card
  if (part.state === 'output-available') {
    const Card = TOOL_CARD_MAP[part.toolName]
    if (!Card) {
      // Unknown tool — render raw JSON as fallback
      return (
        <pre className="text-xs text-text-secondary font-mono bg-bg-elevated/50 rounded-lg p-2 overflow-x-auto">
          {JSON.stringify(part.output, null, 2)}
        </pre>
      )
    }
    return <Card data={part.output} />
  }
  return null
}
```
